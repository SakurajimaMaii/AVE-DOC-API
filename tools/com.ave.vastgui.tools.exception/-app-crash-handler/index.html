App global exception handling.

Here is a usage example:

// App.kt
class App : Application() {

    private val mLogger = ...

    override fun onCreate() {
        super.onCreate()
        ...
        setDefaultUncaughtExceptionHandler { t, e, stackTraceInfo ->
            mLogger.e(stackTraceInfo)
        }
    }

}
